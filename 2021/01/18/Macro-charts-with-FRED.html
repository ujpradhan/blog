Plotting macro data from FRED

  1. Rates: Fed Funds, 2-year, 10-year Treasury, 30-year Treasury
  2. Equity: SPX, NASDAQ
  3. Commodity: Brent, WTI, Natural gas, Copper
  4. Currencies: EUR, GBP, JPY, CNY
import pandas as pd #importing pandas module

import matplotlib.pyplot as plt #importing matplotlib for plotting charts

%matplotlib inline

import matplotlib.dates as mdates

Using fredapi in order to access FRED database. Courtesy of Mortada Meyar.

For installation:

pip install fredapi

from fredapi import Fred 

#fred = Fred("use personal FRED API") 

begin = '2017-12-31'
end = '2021-01-15'

fedfunds = fred.get_series('DFF',
                           observation_start = begin,
                           obervation_end = end) #daily

GT2 = fred.get_series('DGS2',
                      observation_start = begin,
                      obervation_end = end) #daily

GT10 = fred.get_series('DGS10',
                     observation_start = begin,
                     obervation_end = end) #daily

GT30 = fred.get_series('DGS30',
                     observation_start = begin,
                     obervation_end = end) #daily

spx = fred.get_series('SP500',
                     observation_start = begin,
                     obervation_end = end) #daily

nasdaq = fred.get_series('NASDAQCOM',
                     observation_start = begin,
                     obervation_end = end) #daily

brent = fred.get_series('DCOILBRENTEU',
                        observation_start = begin,
                        obervation_end = end) #daily

wti = fred.get_series('DCOILWTICO',
                      observation_start = begin,
                      obervation_end = end) #daily

hh = fred.get_series('DHHNGSP',
                      observation_start = begin,
                      obervation_end = end) #daily

copper = fred.get_series('PCOPPUSDM',
                         observation_start = begin,
                         obervation_end = end) #monthly

usd_eur = fred.get_series('DEXUSEU',
                     observation_start = begin,
                     obervation_end = end) #daily

usd_gbp = fred.get_series('DEXUSUK',
                     observation_start = begin,
                     obervation_end = end) #daily

yen_usd = fred.get_series('DEXJPUS',
                     observation_start = begin,
                     obervation_end = end) #daily

yuan_usd = fred.get_series('DEXCHUS',
                     observation_start = begin,
                     obervation_end = end) #daily

Rates 

fig, axes = plt.subplots(2,2, figsize = (12, 8))

# rotate and align x-labels so they look better
fig.autofmt_xdate()

axes[0, 0].plot(fedfunds)
axes[0, 0].set_title('Effective Fed Funds Rate')
axes[0, 0].set_ylabel('%')


axes[0, 1].plot(GT2)
axes[0, 1].set_title('Generic 2yr Treasury Yield')
axes[0, 1].set_ylabel('%')

axes[1, 0].plot(GT10)
axes[1, 0].set_title('Generic 10yr Treasury Yield')
axes[1, 0].set_ylabel('%')

axes[1, 1].plot(GT30)
axes[1, 1].set_title('Generic 30yr Treasury Yield')
axes[1, 1].set_ylabel('%')

Text(0, 0.5, '%')
